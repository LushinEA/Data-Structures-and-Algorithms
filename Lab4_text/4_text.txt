stack_bracket = []

file = open('../Lab4_text/3_text.txt', encoding='utf-8')

for line in file:
    for sym in line:
        if sym == '(':
            stack_bracket.append(sym)
        elif sym == ')':
            stack_bracket.pop()
file.close()

file_res = open('../Lab4_text/3_answer.txt', 'w', encoding='utf-8')
if len(stack_bracket) == 0:
    print(True)
    file_res.write("True")
else:
    print(False)
    file_res.write("False")
file_res.close()
