from collections import deque
dec_bracket = deque([])

file = open('../Lab4_text/5_text.txt', encoding='utf-8')

for line in file:
    for sym in line:
        if sym == '(':
            dec_bracket.append(sym)
        elif sym == ')':
            dec_bracket.pop()
file.close()

file_res = open('../Lab4_text/5_answer.txt', 'w', encoding='utf-8')
if len(dec_bracket) == 0:
    print(True)
    file_res.write("True")
else:
    print(False)
    file_res.write("False")
file_res.close()
